    function Exo_5_7_innerHTML() {
        //Variables iNombre, i, Somme en Entier
        var iNombre;
        var i;
        var iFacto;
        // Debut
        //Ecrire "Entrez un nombre : "
        //Lire iFacto
        iNombre = parseInt(document.getElementById("iNombreUtilisateur").value);
        //Somme = 0
        iFacto = 1;
        //Pour i = 2 à iNombre
        for (i = 2; i <= iNombre; i++) {
            // iSomme ← iSomme + i
            iFacto = iFacto * i;
        }//fin pour
        //  Ecrire "La factorielle est : ", Facto
        document.getElementById("div_resultat_js").innerHTML = ("innerHTML La factorielle est : "+ iFacto);
    }//fin
