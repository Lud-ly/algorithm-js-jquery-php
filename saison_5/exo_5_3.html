        function Exo_5_3_innerHTML() {
            //DEBUT
            //Variable iNum en numerique
            var iNum = parseInt(document.getElementById("iNombreUtilisateur").value);
            // variable iNum_suivants en numerique
            var iNum_suivants = iNum + 10;
            // Ecrire "Les 10 nombres suivants sont : "
            var sHTML = "Les 10 prochains nombres sont : ";
            // Tant que iNum < iNum_suivants
            while (iNum < iNum_suivants) {
                sHTML += " " + (iNum + 1) + " ";
                iNum = iNum + 1;
                //Fin Tant que
            }
            document.getElementById("div_resultat_js").innerHTML = "innerHTML " + sHTML;
