Exercice 4.2

Cet algorithme est destiné à prédire l'avenir, et il doit être infaillible ! Il lira au clavier l’heure et les minutes, et il affichera l’heure qu’il sera une minute plus tard. Par exemple, si l'utilisateur tape 21 puis 32, l'algorithme doit répondre : "Dans une minute, il sera 21 heure(s) 33". NB : on suppose que l'utilisateur entre une heure valide. Pas besoin donc de la vérifier.

    function Exo_4_2_innerHTML() {
        // Variables heures, minutes en Numérique
        //Début
        var iMinutes;
        var iHeures;
        //Ecrire "Entrez les heures, puis les minutes : "
        //Lire heures,minutes
        
        iHeures = parseInt(document.getElementById("iHeures").value);
        iMinutes =  parseInt(document.getElementById("iMinutes").value);
        //Si minutes = 60 Alors
        console.log(typeof iMinutes);
        iMinutes += 1;
        if (iMinutes == 60) {
            iMinutes = 0;
            iHeures += 1;
            //minutes = 0 et heures = heures+1
        }
        // Si heures = 24 Alors
        if (iHeures == 24) {
            // heures = 0;
            iHeures = 0;
        }
        //ecrire la nouvelle heure
        document.getElementById("div_resultat_js").innerHTML = ("Merci innerHTML, Dans une minute il sera " +  iHeures + " heure(s) " + iMinutes + " minute(s)");
    }
