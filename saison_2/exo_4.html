ALGORITHME ET PSEUDO-CODE:

Exercice 2.4

Ecrire un algorithme utilisant des variables de type chaîne de caractères, et affichant quatre variantes possibles de la célèbre « belle marquise, vos beaux yeux me font mourir d’amour ». On ne se soucie pas de la ponctuation, ni des majuscules.

Pseudocode:

VARIABLES string sOne = " Belle marquise "
VARIABLES string sTwo = " Vos beaux yeux "
VARIABLES string sThree = " Me font "
VARIABLES string sFour = " Mourir d’amour
VARIABLES VARIANT,VARIANT1,VARIANT2,VARIANT3,VARIANT4 EN STRING
DEBUT
SI VARIANT1 ALORS
ECRIRE sOne+sTwo+sThree+sFour
SI VARIANT2 ALORS
ECRIRE sTwo+sOne+sFour+sThree
SI VARIANT3 ALORS
ECRIRE sOne+sTwo+sFour+sThree
ECRIRE sTwo+sThree+sOne+sFour
SI VARIANT4 ALORS
ECRIRE sFour + sOne + sThree + sTwo
FIN

Tester l'algorithme :

                                   function Exo_2_4_innerHTML() {
                //VARIABLES VARIANT,VARIANT1,VARIANT2,VARIANT3,VARIANT4 EN STRING
                var sOne = " Belle marquise "
                var sTwo = " Vos beaux yeux "
                var sThree = " Me font "
                var sFour = " Mourir d’amour "
                //DEBUT
                Variante = document.getElementById("Variante").value;
                //SI VARIANT0 ALORS
                //ECRIRE one+two+three+four
                if (Variante == 0) {
                    document.getElementById("div_resultat_js").innerHTML = ("innerHTML-> " + sOne + sTwo + sThree + sFour);
                }
                //SI VARIANT1 ALORS
                //ECRIRE one+two+three+four
                if (Variante == 1) {
                    document.getElementById("div_resultat_js").innerHTML = ("innerHTML-> " + sThree + sTwo + sOne + sFour);
                }
                //SI VARIANT2 ALORS
                else if (Variante == 2) {
                    //ECRIRE two+one+three+four
                    document.getElementById("div_resultat_js").innerHTML = ("innerHTML-> " + sTwo + sOne + sThree + four);
                }
                //SI VARIANT3 ALORS
                else if (Variante == 3) {
                    //ECRIRE one+two+four+three
                    document.getElementById("div_resultat_js").innerHTML = ("innerHTML-> " + sOne + sTwo + sFour + sThree);
                }
                //ECRIRE two+three+one+four
                //SI VARIANT4 ALORS
                else if (Variante == 4) {
                    //ECRIRE four + one + three + two
                    document.getElementById("div_resultat_js").innerHTML = ("innerHTML-> " + sFour + sOne + sThree + sTwo);
                }
                //FIN
            }
