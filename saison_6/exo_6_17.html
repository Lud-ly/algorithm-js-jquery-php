    function tri_insertion(tab) {
        //Variables iValeur,j,i en Numérique
        var iValeur, j, i;
        //Pour i←0 à longueur de tab
        for (i = 0; i < tab.length; i++) {
            //on prends la valeur à l'index i
            iValeur = tab[i];
            // index de départ 0
            //j←i
            j = i;
            //Tant que j est supérieur à 0 et tab[j]est supérieur à iValeur alors
            while (j > 0 && tab[j - 1] > iValeur) {
                //On décale j à j-1
                tab[j] = tab[j - 1];
                //j←j-1
                j = j - 1;
            }//Fin de tant que
            //i Suivant

            //tab[j]iValeur
            tab[j] = iValeur;
        }
    }
