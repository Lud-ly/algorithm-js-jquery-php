Exercice 6.14

Toujours et encore sur le même principe, écrivez un algorithme permettant, à l’utilisateur de saisir les notes d'une classe. Le programme, une fois la saisie terminée, renvoie le nombre de ces notes supérieures à la moyenne de la classe.

Pseudocode:

Variables Nb, i, Som, Moy, Nbsup en Numérique
Tableau T() en Numérique
Debut
Ecrire "Entrez le nombre n° ", i + 1
Ecrire "Entrer les notes"
Lire value Nb
Ecrire "Entrez le nombre de notes à saisir : "
Lire value myVal
pusher myVal dans array T
calc somme T et moyenne moy
Pour chaque val dans T
Si val > moy
nbSup = nbSup + 1;
FinSi
Ecrire "Les notes ,T"
Ecrire "la moyenne est de : " , moy
Ecrire NbSup, " élèves dépassent la moyenne de la classe" arrondir
Fin

Tester l'algorithme :

Sous le capot

                                    JAVASCRIPT
                                     
                var T = [];
                var i = 2;
                var nb;
                var moy = 0;
                var som = 0;
                var myVal = 0;
    
                function Exo_6_14_innerHTML() {
                    // Variables Nb, i, Som, Moy, Nbsup en Numérique
                    //Tableau T() en Numérique
                    //Debut
                    //Ecrire "Entrez le nombre n° ", i + 1
                    document.getElementById("resultat_js").innerHTML = ("Entrez le nombre n° " + i++);
                    //Ecrire "Entrer les notes"
                    //Lire value Nb
                    nb = parseInt(document.getElementById("nombreP").value);
                    //Ecrire "Entrez le nombre de notes à saisir : "
                    // Lire value myVal
                    myVal = (parseInt(document.getElementById("nombrePlat").value));
                    //pusher myVal dans array T
                    T.push(myVal);
                    //calc somme T et moyenne moy
                    som += myVal;
                    moy = som / T.length;
                    //Pour chaque val dans T
                    T.forEach(function (val) {
                        //Si val > moy
                        if (val > moy) {
                            // nbSup = nbSup + 1;
                            nbSup++;
                        }// FinSi
                    });
                    if (nb == T.length) {
                        //Ecrire "Les notes ,T"
                        document.getElementById("div_result").innerHTML = ("innerHTML voici les notes : " + T);
                        //Ecrire "la moyenne est de : " , moy
                        document.getElementById("div_resultat").innerHTML = ("la moyenne est de : " + moy.toFixed(2));
                        //Ecrire NbSup, " élèves dépassent la moyenne de la classe" arrondir
                        document.getElementById("div_resultat_js").innerHTML = (nbSup + " élèves dépassent la moyenne de la classe");
                    }
                } //Fin
