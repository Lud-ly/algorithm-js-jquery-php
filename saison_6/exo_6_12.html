Exercice 6.12

Ecrivez un algorithme qui permette la saisie d’un nombre quelconque de valeurs, sur le principe de l’ex 6.8. Toutes les valeurs doivent être ensuite augmentées de 1, et le nouveau tableau sera affiché à l’écran.

Pseudocode:

Variables Nb, Nbpos, Nbneg en Numérique
Variables global Tableau T() ,i en Numérique
Début
Ecrire "Entrez le nombre n° ", i + 1
Lire T(value)
Pour i dans longueur tableau
T[i]+=1
Ecrire "Entrez le nombre de valeurs :"
Lire Nb
Si nb = longueur de T alors
Ecrire Notes
FinSi
Pacourir valeur de T
Si val inferieur 0 alors
Nbneg = Nbneg + 1
Nbpos = Nbpos + 1
Finsi
Ecrire Nombre de valeurs positives : , Nbpos
Ecrire Nombre de valeurs négatives : , Nbneg
Fin

var i = 2;
var T = [];
function Exo_6_12_innerHTML() {
    // Variables Nb, Nbpos, Nbneg en Numérique
    //Variables global Tableau T() ,i en Numérique
    var Nbpos = 0;
    var Nbneg = 0;
    var nb;
    //  Ecrire "Entrez le nombre n° ", i + 1 
    // Lire T(value)
    document.getElementById("div_resultat").innerHTML = ("Entrer la notes n°" + [i++]);

    T.push(parseInt(document.getElementById("nombreP").value));
    //pour i dans longueur tableau
    //T[i]+=1
    for (i = 0; i < T.length; i++) {
        T[i] += 1;
    }
    // Ecrire "Entrez le nombre de valeurs :"
    //  Lire Nb
    nb = (parseInt(document.getElementById("nombrePlat").value));

    //Si nb = longueur de T alors
    if (nb == T.length) {
        //Ecrire Notes
        document.getElementById("div_resultat_js").innerHTML = T.join(" , ");
        document.getElementById("div_resultat").innerHTML = ("innerHTML Voici le nouveau tableau : ");
    }//FinSi
    console.log(T);
    //pacourir valeur de T
    T.forEach(function (val) {
        //Si val inferieur 0 alors
        if (val < 0) {
            // Nbneg = Nbneg + 1
            Nbneg++;
        } else {
            // Nbpos = Nbpos + 1
            Nbpos++;
        }
    });
    //Finsi
    // Ecrire "Nombre de valeurs positives : ", Nbpos                           
    //Ecrire "Nombre de valeurs négatives : ", Nbneg 
    document.getElementById("div_result").innerHTML = ("Nombre de valeurs positives : " + Nbpos + "
" + " Nombre de valeurs négatives : " + Nbneg);
