Ecrire un algorithme permettant de résoudre le problème suivant :
– Données : un tableau contenant 100 entiers
– Résultat : “vrai” si les entiers sont consécutifs et “faux” sinon.

    function Exo_6_16_innerHTML() {
        //tab result en Numérique
        var results = [];
        //Pour i de 0 à 100
        //Generer 100 nombres de 0 à 100
        for (var i = 0; i <= 100; i++) {
            var nbrAlea = Math.floor(100 * Math.random());
            //Ajouter dans array
            results += nbrAlea + " , ";
        }
        //Si result[i]>result[i+1]
        if (results[i] > results[i + 1]) {
            //Ecrire Consecutif
            document.getElementById("div_result_js").innerHTML = ("innerHTML consecutif");
        }
        //Sinon pas Consecutif
        else {
            document.getElementById("div_resultat").innerHTML = ("innerHTML Pas consecutif");
        }
        //Ecrire results
        document.getElementById("div_result").innerHTML = (results);
    }
