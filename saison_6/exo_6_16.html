Exercice 6.16

Ecrire un algorithme permettant de résoudre le problème suivant :
– Données : un tableau contenant 100 entiers
– Résultat : “vrai” si les entiers sont consécutifs et “faux” sinon.

Pseudocode:

Tester l'algorithme :

Sous le capot

                                    JAVASCRIPT
                                     
var results = [];

    function Exo_6_16_innerHTML() {
        //Variable tableau results en Numérique
        //Variables nbrAlea,i en Numérique
        //Si j inférieur 100 alors
        var i;
        results.push(parseInt(document.getElementById("myVal").value));
        //Pour i de 0 à longueur de results
        for (i = 0; i < results.length; i++) {

            if (i > 1) {
                //Si results[i]=results[i+1]-1
                if (results[i] == results[i - 1] + 1) {
                    //Ecrire Consecutif
                    document.getElementById("div_resultat").innerHTML = ("innerHTML Les nombres sont consecutif");
                }//FinSi
                //Sinon pas Consecutif
                else {
                    document.getElementById("div_resultat").innerHTML = ("innerHTML Les nombres ne sont pas consecutif");
                }
            }//FinSi
        }//i suivant
        //Ecrire results
        document.getElementById("div_result").innerHTML = (results);
    }
