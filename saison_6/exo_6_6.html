ALGORITHME ET PSEUDO-CODE:

Exercice 6.6

Que produit l’algorithme suivant ?

Tableau Suite(7) en Entier
Variable i en Entier
Début
Suite(0) ← 1
Suite(1) ← 1
Pour i ← 2 à 7
Suite(i) ← Suite(i-1) + Suite(i-2)
i suivant
Pour i ← 0 à 7
Ecrire Suite(i)
i suivant
Fin

Tester l'algorithme :

Sous le capot

                                    JAVASCRIPT
                                     
    function Exo_6_6_innerHTML() {
        // Tableau Suite(7) en Entier
        //Variable i en Entier
        // Début
        var suite = [];
        var i = 2;
        // Suite(0) ← 1
        // Suite(1) ← 1
        document.getElementById("div_resultat_js").innerHTML = (suite[0] = 1);
        document.getElementById("div_resultat_js").innerHTML = (suite[1] = 1);
        //Pour i ← 2 à 7
        for (i = 2; i <= 7; i++) {
            //Suite(i) ← Suite(i-1) + Suite(i-2)
            suite[i] = suite[i - 1] + suite[i - 2];
            document.getElementById("div_resultat_js").innerHTML = (suite[i]);
        }
        document.getElementById("div_resultat_js").innerHTML = ("innerHTML " + suite);

    } //Fin
