Exercice 8.6

Soit un tableau T à deux dimensions (12, 8) préalablement rempli de valeurs numériques. Écrire un algorithme qui recherche la plus grande valeur au sein de ce tableau.

    function Exo_8_6_innerHTML() {
        // Variables i, j, iMax, jMax en Numérique
        // Tableau T(12, 8) en Numérique
        var i;
        var j;
        var iMax = 0;
        var jMax = 0;
        var iMin = 0;
        var jMin = 0;
        var T = [[10, 50, 16, 152, 55, 9, 24, 30, 124, 25, 15, 32], [70, 8, 50, 23, 54, 22, 51, 60]];
        document.getElementById("div_resultat_js").innerHTML = ("Tableau 2dim: " + "[[" + T[0] + "],[" + T[1] + "]]");
        // Debut
        // Pour i ← 0 à 12
        for (i = 0; i < T.length; i++) {
            // Pour j ← 0 à 8
            for (j = 0; j < T[1].length; j++) {
                // Si T(i, j) > T(iMax, jMax) Alors
                if (T[i][j] > T[iMax][jMax]) {
                    // iMax ← i
                    // jMax ← j
                    iMax = i;
                    jMax = j;
                } // FinSi
                if (T[i][j] < T[iMin][jMin]) {
                    // iMin ← i
                    // jMin ← j
                    iMin = i;
                    jMin = j;
                } // FinSi
            }// j Suivant
        }// i Suivant   

        // Ecrire "Le plus grand élément est ", T(iMax, jMax)
        // Ecrire "Il se trouve aux indices ", iMax, "; ", jMax
        document.getElementById("div_result_js").innerHTML = ("Le plus grand élément est " + T[iMax][jMax] + ", le plus petit élément est " + T[iMin][jMin]);
        document.getElementById("div_result").innerHTML = ("Le grand se trouve dans le tableau " + (iMax + 1) + " position " + (jMax + 1) + ", le petit se trouve dans le tableau " + (iMin + 1) + " Position " + (jMin + 1));
    } // Fin
