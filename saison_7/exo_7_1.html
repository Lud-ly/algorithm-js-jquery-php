Exercice 7.1

Ecrivez un algorithme qui permette de saisir un nombre quelconque de valeurs, et qui les range au fur et à mesure dans un tableau. Le programme, une fois la saisie terminée, doit dire si les éléments du tableau sont tous consécutifs ou non. Par exemple, si le tableau est : 12 13 14 15 16 17 18 ses éléments sont tous consécutifs. En revanche, si le tableau est : 9 10 11 15 16 17 18 ses éléments ne sont pas tous consécutifs.

Pseudocode:

Tester l'algorithme :

Sous le capot

    //Variables Tableau en Numérique
    //Variables Nb, i en Entier
    //Variable Flag en Booleen
    var i = 2;
    var T = [];
    var nb = 0;
    var Flag = false;

    function Exo_7_1_innerHTML() {
        //Début
        //Ecrire Entrer la notes n°"+i suivant
        document.getElementById("div_resultat_js").innerHTML = ("Entrer la notes n°" + [i++]);
        //Ajouter value à notes
        T.push(parseInt(document.getElementById("iVal").value));
        //Si longueur Tab >=1 ET <=5 alors
        if (T.length >= 1 && T.length <= 5) {
            //Ecrire tab Notes
            document.getElementById("div_resultat").innerHTML = T.join(" , ");
        }//FinSi
        //Flag ← Vrai
        Flag = true;
        //Pour i ← 1 à longueur Tab
        for (i = 1; i < T.length; i++) {
            // Si T(i) <> T(i – 1) + 1 Alors
            if (T[i] < T[i - 1] + 1) {
                // Flag ← Faux
                Flag = false;
                // FinSi

            }
            // Si Flag Alors
            if (Flag) {
                //Ecrire "Les nombres sont consécutifs"
                document.getElementById("div_result").innerHTML = (T + "innerHTML Les nombres sont consécutifs");

            }//FinSi
            // Sinon
            else {
                //Ecrire "Les nombres ne sont pas consécutifs"
                document.getElementById("div_result").innerHTML = (T + " Les nombres ne sont pas tous consécutifs");
            }
        }//i Suivant
    }//Fin
