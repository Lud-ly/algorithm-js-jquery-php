    function Exo_9_5() {
        var phrase = "";
        var charDelete = 0;
        var newPhrase = "";
        phrase = String(prompt("Entrez une phrase : "));

        charDelete = parseInt(prompt("Entrez le rang du caractère à supprimer : "));

        phrase = phrase.substring(0, charDelete) + phrase.substring(charDelete + 1);
        alert(phrase);
        newPhrase = String(prompt("Entrez la nouvelle phrase : "));

        phrase = phrase.replace(phrase, newPhrase);
        alert("Voici la nouvelle phrase" + phrase);
        }
