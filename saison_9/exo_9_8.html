function Exo_9_8_innerHTML() {
    var jouer = document.getElementById("myVal").value;

    while (jouer) {

        //#3 tirages
        var tirage = [];
        tirage.push(Math.floor(Math.random() * 6) + 1);  //entre 1 et 6
        console.log(tirage);
        tirage.push(Math.floor(Math.random() * 6) + 1);
        tirage.push(Math.floor(Math.random() * 6) + 1);  //idem ci-dessus toujours entre 1 et 6

        //affichage du tirage et des points obtenus
        var points = 0;
        var i = 0;
        while (i <= 2) {
            points = points + tirage[i];
            i = i + 1;

        }

        document.getElementById("div_resultat").innerHTML = (tirage);
        document.getElementById("div_resultat_js").innerHTML = ("Vous avez gagné " + (points) + " points à ce lancer");
        var rep = document.getElementById("restart").value;
        if (rep != "o" && rep != "O") {
            jouer = false;
        }
    }
}
