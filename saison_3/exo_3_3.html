    function Exo_3_3_innerHTML() {
        //sNom,sNom2,sNom3 en string
        sNom = "";
        sNom2 = "";
        sNom3 = "";
        //Début
        //Ecrire "entrez 3 noms" 
        //ecrire"donnez le nom n°1 
        //ecrire "donnez le nom n°2 
        //ecrire"donnez le nom n°3 
        //lire sNom,sNom2,sNom3
        sNom = document.getElementById("sNom").value;
        sNom2 = document.getElementById("sNom2").value;
        sNom3 = document.getElementById("sNom3").value;
        //Si sNom < sNom2 ET sNom2 < sNom3 Alors 
        if (sNom < sNom2 && sNom2 < sNom3) {
            //Ecrire "Ces noms sont classés alphabétiquement"
            document.getElementById("div_resultat_js").innerHTML = "Bravo innerHTML  Ces noms sont classés alphabétiquement !";
        }
        //Sinon 
        //Ecrire "Ces noms ne sont pas classés" 
        else {
            document.getElementById("div_resultat_js").innerHTML = "Désolé innerHTML Ces noms ne sont pas classés";
        }//FinSi
    }//Fin
