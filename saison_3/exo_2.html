Exercice 3.2

Ecrire un algorithme qui demande deux nombres à l’utilisateur et l’informe ensuite si leur produit est négatif ou positif (on laisse de côté le cas où le produit est nul). Attention toutefois : on ne doit pas calculer le produit des deux nombres.

Pseudocode:

nombre1 en numerique
nombre2 en numerique
debut
ecrire nombre1="donnez un nombre n°1 "
ecrire nombre2="donnez un nombre n°2 "
lire nombre1 et nombre2
si nombre1 ET nombre2 >0 alors
ecrire le nombre est positif
sinonsi nombre1 ET nombre2 inferieur 0 alors
ecrire le nombre est negatif
fin

                                        function Exo_3_2_innerHTML() {
                                            // DEBUT
                                            //nombre en numerique
                                            //nombre2 en numerique
                                            var iNombre;
                                            var iNombre2;
                                            //DEBUT
                                            // ecrire nombre1="donnez un nombre n°1 
                                            // ecrire nombre2="donnez un nombre n°2 
                                            //lire nombre1 et nombre2
                                            iNombre = document.getElementById("iNombreUtilisateur").value;
                                            iNombre2 = document.getElementById("iNombreUtilisateur2").value;
                                            // Si Nombre est supérieur à 0 alors
                                            if (iNombre > 0 && iNombre2 > 0) {
                                                //  Ecrire le Nombre est positif
                                                document.getElementById("div_resultat_js").innerHTML = "Bravo innerHTML le produit est positif !";
                                            }
                                            // sinonsi iNombre < 0 alors
                                            else if(iNombre < 0 && iNombre2 < 0) {
                                                document.getElementById("div_resultat_js").innerHTML = "Bravo innerHTML le produit est negatif !";
                                            }
                                            //Fin
                                        }
